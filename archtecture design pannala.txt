Marshaled schema query retrieval for Kudu database
Idho step 1 SQL — Supabase SQL Editor la full-a copy-paste pannunga:

sql
-- Roles & permissions (data-driven, supports custom roles)
create table roles (
  id uuid primary key default gen_random_uuid(),
  name text not null unique,
  is_custom boolean not null default false,
  created_at timestamptz not null default now()
);

create table permissions (
  id uuid primary key default gen_random_uuid(),
  key text not null unique  -- e.g. 'view_orders', 'edit_menu'
);

create table role_permissions (
  role_id uuid references roles(id) on delete cascade,
  permission_id uuid references permissions(id) on delete cascade,
  primary key (role_id, permission_id)
);

-- Staff users (linked to Supabase Auth)
create table staff_users (
  id uuid primary key default gen_random_uuid(),
  auth_id uuid references auth.users(id) on delete set null,
  name text not null,
  phone text,
  role_id uuid references roles(id),
  is_active boolean not null default true,
  created_at timestamptz not null default now()
);

-- Tables + QR mapping
create table restaurant_tables (
  id uuid primary key default gen_random_uuid(),
  table_no int not null unique,
  qr_code_url text,
  assigned_waiter_id uuid references staff_users(id)
);

-- Menu
create table categories (
  id uuid primary key default gen_random_uuid(),
  name text not null,
  sort_order int not null default 0
);

create table menu_items (
  id uuid primary key default gen_random_uuid(),
  category_id uuid references categories(id) on delete cascade,
  name text not null,
  price numeric(10,2) not null,
  image_url text,
  is_available boolean not null default true
);

-- Orders
create table orders (
  id uuid primary key default gen_random_uuid(),
  table_id uuid references restaurant_tables(id),
  customer_name text not null,
  customer_phone text not null,
  status text not null default 'pending'
    check (status in ('pending','confirmed','preparing','ready','served','billed')),
  waiter_id uuid references staff_users(id),
  created_at timestamptz not null default now()
);

create table order_items (
  id uuid primary key default gen_random_uuid(),
  order_id uuid references orders(id) on delete cascade,
  menu_item_id uuid references menu_items(id),
  qty int not null check (qty > 0),
  notes text,
  price_at_order numeric(10,2) not null  -- snapshot price, protects against later price changes
);

create table order_status_history (
  id uuid primary key default gen_random_uuid(),
  order_id uuid references orders(id) on delete cascade,
  status text not null,
  changed_by uuid references staff_users(id),
  changed_at timestamptz not null default now()
);

create table bills (
  id uuid primary key default gen_random_uuid(),
  order_id uuid references orders(id) unique,
  total numeric(10,2) not null,
  cashier_id uuid references staff_users(id),
  paid_at timestamptz
);

-- Seed base roles + permissions
insert into roles (name, is_custom) values
  ('admin', false), ('waiter', false), ('kitchen', false), ('cashier', false);

insert into permissions (key) values
  ('view_orders'),('edit_orders'),('confirm_orders'),('view_kitchen_queue'),
  ('update_prep_status'),('view_billing'),('generate_bills'),('edit_menu'),
  ('manage_staff'),('manage_roles'),('view_revenue'),('export_data'),('manage_tables');

-- Wire up default role permissions (adjust as needed)
insert into role_permissions (role_id, permission_id)
select r.id, p.id from roles r, permissions p
where r.name = 'admin'; -- admin gets everything

insert into role_permissions (role_id, permission_id)
select r.id, p.id from roles r, permissions p
where r.name = 'waiter' and p.key in ('view_orders','edit_orders','confirm_orders');

insert into role_permissions (role_id, permission_id)
select r.id, p.id from roles r, permissions p
where r.name = 'kitchen' and p.key in ('view_kitchen_queue','update_prep_status');

insert into role_permissions (role_id, permission_id)
select r.id, p.id from roles r, permissions p
where r.name = 'cashier' and p.key in ('view_billing','generate_bills');

-- Enable RLS on every table
alter table roles enable row level security;
alter table permissions enable row level security;
alter table role_permissions enable row level security;
alter table staff_users enable row level security;
alter table restaurant_tables enable row level security;
alter table categories enable row level security;
alter table menu_items enable row level security;
alter table orders enable row level security;
alter table order_items enable row level security;
alter table order_status_history enable row level security;
alter table bills enable row level security;

-- Public read access for menu (customers browsing don't log in)
create policy "public can view categories" on categories for select using (true);
create policy "public can view available menu items" on menu_items for select using (is_available = true);

-- Public (anonymous) customers can create orders, but not edit/delete them
create policy "public can create orders" on orders for insert with check (true);
create policy "public can view own order by id" on orders for select using (true);
create policy "public can add order items on creation" on order_items for insert with check (true);
create policy "public can view order items" on order_items for select using (true);

-- Staff policies: helper function checking role_permissions via the authenticated user's staff_users row
create or replace function has_permission(perm_key text)
returns boolean language sql security definer as $$
  select exists (
    select 1 from staff_users su
    join role_permissions rp on rp.role_id = su.role_id
    join permissions p on p.id = rp.permission_id
    where su.auth_id = auth.uid() and p.key = perm_key and su.is_active = true
  );
$$;

create policy "staff with edit_orders can update orders" on orders
  for update using (has_permission('edit_orders') or has_permission('confirm_orders'));

create policy "staff with view_orders can view all orders" on orders
  for select using (has_permission('view_orders'));

create policy "staff with edit_menu can manage menu" on menu_items
  for all using (has_permission('edit_menu'));

create policy "staff with manage_staff can manage staff" on staff_users
  for all using (has_permission('manage_staff'));

create policy "staff with manage_roles can manage roles" on roles
  for all using (has_permission('manage_roles'));

create policy "staff with generate_bills can manage bills" on bills
  for all using (has_permission('generate_bills') or has_permission('view_billing'));
